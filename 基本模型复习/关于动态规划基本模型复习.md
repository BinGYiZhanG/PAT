#### 11.3 最长不下降子序列(非连续)
```cpp
#include <bits/stdc++.h>

using namespace std;

const int N=100;

int A[N],dp[N];

int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&A[i]);
    }

    int ans=-1;
    for(int i=1;i<=n;i++){
        dp[i]=1;///先假设每个元素自成一个子序列
        for(int j=1;j<i;j++){
            if(A[i]>=A[j] && (dp[j]+1>dp[i])){
                dp[i]=dp[j]+1;
            }
        }
        ans=max(ans,dp[i]);
    }
    printf("%d",ans);
    return 0;
}

```
#### 11.4 最长公共子序列
#### 注：求最长公共子序列时，只能倒着遍历字符串A,B
```cpp
#include <bits/stdc++.h>

using namespace std;

const int N=100;

char A[N],B[N];
int dp[N][N];
string ans="";

int main(){
//    int n;
    gets(A+1);//从下标为1开始读入
    gets(B+1);

    int lenA=strlen(A+1);//因为读入时下标从1开始，因此读取长度也从+1开始
    int lenB=strlen(B+1);

    for(int i=0;i<=lenA;i++){
        dp[i][0]=0;
    }

    for(int j=0;j<=lenB;j++){
        dp[0][j]=0;
    }

    for(int i=1;i<=lenA;i++){
        for(int j=1;j<=lenB;j++){
            if(A[i]==B[j]){
                dp[i][j]=dp[i-1][j-1]+1;
            }
            else{
                dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
            }
        }
    }
    ///求最长公共子序列
    int i=lenA,j=lenB;
    while(i>0&&j>0){
        if(A[i]==B[j]){
            ans+=A[i];
            i--;
            j--;
        }
        else if(dp[i-1][j]<dp[i][j-1]){
            j--;///表示不录入第二个序列的第j个字符时的最长公共子序列
        }
        else if(dp[i-1][j]>=dp[i][j-1]){
            i--;///表示不录入第一个序列的第i个字符时的最长公共子序列，
        }
    }
    printf("%d\n",dp[lenA][lenB]);
    cout<<ans<<endl;
    return 0;
}
```
